Arduino Code — Full Example

This example uses SoftwareSerial to talk with HC-05 so you can keep hardware Serial for debug. It reads single-character commands from phone: F,B,L,R,S (forward, back, left, right, stop).

#include <SoftwareSerial.h>

SoftwareSerial BT(10, 11); // RX, TX for HC-05 (change if needed)

// Motor A (left)
const int ENA = 5;   // PWM pin
const int IN1 = 4;
const int IN2 = 3;

// Motor B (right)
const int ENB = 6;   // PWM pin
const int IN3 = 8;
const int IN4 = 7;

void setup() {
  // serial
  Serial.begin(9600);
  BT.begin(9600);

  // motor pins
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);

  pinMode(ENB, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  stopMotors();
}

void loop() {
  if (BT.available()) {
    char cmd = BT.read();
    Serial.print("Cmd: ");
    Serial.println(cmd);
    executeCommand(cmd);
  }
}

void executeCommand(char c) {
  c = toupper(c);
  if (c == 'F') forward();
  else if (c == 'B') backward();
  else if (c == 'L') turnLeft();
  else if (c == 'R') turnRight();
  else if (c == 'S') stopMotors();
}

void forward() {
  analogWrite(ENA, 200);
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);

  analogWrite(ENB, 200);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void backward() {
  analogWrite(ENA, 200);
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);

  analogWrite(ENB, 200);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, HIGH);
}

void turnLeft() {
  // left wheel slow/stop, right wheel forward
  analogWrite(ENA, 0);
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);

  analogWrite(ENB, 200);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void turnRight() {
  // right wheel slow/stop, left wheel forward
  analogWrite(ENB, 0);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);

  analogWrite(ENA, 200);
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
}

void stopMotors() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

Line-by-line notes

- SoftwareSerial BT(10,11) creates a serial on pins 10(RX) and 11(TX) — adjust to your wiring.
- Motor pins: INx control polarity (direction), ENx gets PWM (speed). analogWrite(0..255).
- Commands are single chars for simplicity; you can extend to commands like "SPEED:180".
