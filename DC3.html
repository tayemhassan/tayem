int ENA = 9;   // PWM pin for speed
int IN1 = 8;   // Direction pin 1
int IN2 = 7;   // Direction pin 2

void setup() {
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
}

void loop() {
  // Forward
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  analogWrite(ENA, 200); // Speed (0-255)
  delay(2000);

  // Stop
  analogWrite(ENA, 0);
  delay(1000);

  // Backward
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, HIGH);
  analogWrite(ENA, 150);
  delay(2000);

  // Stop
  analogWrite(ENA, 0);
  delay(1000);
}
